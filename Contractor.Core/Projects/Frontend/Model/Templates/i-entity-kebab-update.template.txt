import { ApiEntityUpdate } from './api/api-entity-kebab-update';
import { IEntityDetail } from './i-entity-kebab-detail';

export interface IEntityUpdate {
    id: string;
}

export function toApiEntityUpdate(entityUpdate: IEntityUpdate): ApiEntityUpdate {
    return {
        id: entityUpdate.id,
    };
}

export function fromEntityDetail(entityDetail: IEntityDetail): IEntityUpdate {
    return {
        id: entityDetail.id,
    };
}
