using ProjectName.Contract.Persistence.Modules.Domain.Entities;
using System;

namespace ProjectName.Persistence.Modules.Domain.Entities
{
    internal class DbEntity : IDbEntity
    {
        public Guid Id { get; set; }

        internal static void UpdateEfEntity(EfEntity efEntity, IDbEntity dbEntity)
        {
        }

        internal static IDbEntity FromEfEntity(EfEntity efEntity)
        {
            if (efEntity == null)
            {
                return null;
            }

            return new DbEntity()
            {
                Id = efEntity.Id,
            };
        }

        internal static EfEntity ToEfEntity(IDbEntity dbEntity, Guid requestScopeId)
        {
            return new EfEntity()
            {
                Id = dbEntity.Id,
                RequestScopeId = requestScopeId,
            };
        }
    }
}
