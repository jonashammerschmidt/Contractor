using ProjectName.Contract.Persistence.Modules.Domain.Entities;
using System;

namespace ProjectName.Persistence.Modules.Domain.Entities
{
    internal class DbEntityListItem : IDbEntityListItem
    {
        public Guid Id { get; set; }

        internal static IDbEntityListItem FromEfEntity(EfEntity efEntity)
        {
            if (efEntity == null)
            {
                return null;
            }

            return new DbEntityListItem()
            {
                Id = efEntity.Id,
            };
        }
    }
}
