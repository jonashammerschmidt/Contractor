using ProjectName.Contract.Persistence.Model.Domain.Entities;
using Microsoft.VisualStudio.TestTools.UnitTesting;
using System;

namespace ProjectName.Persistence.Tests.Model.Domain.Entities
{
    internal class DbEntityTest : IDbEntity
    {
        public Guid Id { get; set; }

        public static IDbEntity Default()
        {
            return new DbEntityTest()
            {
                Id = EntityTestValues.Id,
            };
        }

        public static IDbEntity Updated()
        {
            return new DbEntityTest()
            {
                Id = EntityTestValues.Id,
            };
        }

        public static void AssertWithDefault(IDbEntity dbEntity)
        {
            Assert.AreEqual(EntityTestValues.Id, dbEntity.Id);
        }

        public static void AssertWithUpdated(IDbEntity dbEntity)
        {
            Assert.AreEqual(EntityTestValues.Id, dbEntity.Id);
        }
    }
}
