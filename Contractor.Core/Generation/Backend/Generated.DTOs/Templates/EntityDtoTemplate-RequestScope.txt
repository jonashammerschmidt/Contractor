using DbProjectName.Persistence.DbContext.Modules.Domain.Entities;
using System;

namespace ProjectName.Generated.DTOs.Modules.Domain.Entities
{
    public class EntityDto
    {
        public Guid Id { get; set; }

        public static EntityDto FromEfEntity(EfEntity efEntity)
        {
            if (efEntity == null)
            {
                return null;
            }

            return new EntityDto()
            {
                Id = efEntity.Id,
            };
        }

        public override string ToString()
        {
            return $"{this.DisplayProperty} ({this.Id})";
        }
    }
}
