using DbProjectName.Persistence.DbContext.Modules.Domain.Entities;
using System;

namespace ProjectName.Modules.Domain.Entities
{
    public class EntityDto : EntityDtoData
    {
        public Guid Id { get; set; }

        public Guid RequestScopeId { get; set; }

        public EntityDto() { }

        public EntityDto(EntityDto entityDto)
        {
            this.Id = entityDto.Id;
            this.RequestScopeId = entityDto.RequestScopeId;
        }

        public EntityDto(Guid id, Guid requestScopeId, EntityDtoData entityDtoData)
        {
            this.Id = id;
            this.RequestScopeId = requestScopeId;
        }

        public static EntityDto FromEfEntity(EfEntity efEntity)
        {
            if (efEntity == null)
            {
                return null;
            }

            return new EntityDto()
            {
                Id = efEntity.Id,
                RequestScopeId = efEntity.RequestScopeId,
            };
        }

        public static EntityDto FromEntityDtoData(Guid id, Guid requestScopeId, EntityDtoData entityDtoData)
        {
            return new EntityDto()
            {
                Id = id,
                RequestScopeId = requestScopeId,
            };
        }

        public static EfEntity ToEfEntity(EntityDto entityDto)
        {
            return new EfEntity()
            {
                Id = entityDto.Id,
                RequestScopeId = entityDto.RequestScopeId,
            };
        }

        public override string ToString()
        {
            return $"{this.DisplayProperty} ({this.Id})";
        }
    }
}
