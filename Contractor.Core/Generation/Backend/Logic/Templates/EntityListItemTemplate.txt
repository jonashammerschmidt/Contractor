using ProjectName.Contract.Logic.Modules.Domain.Entities;
using ProjectName.Contract.Persistence.Modules.Domain.Entities;
using System;

namespace ProjectName.Logic.Modules.Domain.Entities
{
    internal class EntityListItem : IEntityListItem
    {
        public Guid Id { get; set; }

        internal static IEntityListItem FromDbEntityListItem(IDbEntityListItem dbEntityListItem)
        {
            if (dbEntityListItem == null)
            {
                return null;
            }

            return new EntityListItem()
            {
                Id = dbEntityListItem.Id,
            };
        }
    }
}
